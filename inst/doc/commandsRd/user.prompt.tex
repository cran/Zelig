\include{zinput} \begin{document}\section{{\tt user.prompt}: Pause in demo files}\label{ss:user.prompt}
\keyword{file}{user.prompt}
\begin{Description}\relax
Use \code{user.prompt} while writing demo files to force users to hit
return before continuing.
\end{Description}
\begin{Usage}
\begin{verbatim}
user.prompt()
\end{verbatim}
\end{Usage}
